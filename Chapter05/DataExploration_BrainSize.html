fig,ax = plt.subplots(1,2,figsize=(10,4))
sns.histplot(data['Neurons'],ax=ax[0])
sns.kdeplot(data['Neurons'],ax=ax[1])
plt.show()

sns.pairplot(data)
plt.show()

# Compute the cross correlation
corr = data.corr()
